# Selectors:
- element
- #id
- .class

# Compound selectors:
- A E: space - all descendants
- A > E: > - child of A
- E:first-child - first child
- B + E, sibling of B
- . - one specific element
- comma - *and*

# Attribute selectors:
- a[=attribute]
- a[~=attribute]

# Specificity:

id = 10
class = 1
